class Solution {
public:
    vector<string> letterCombinations(const string digits) {
        unordered_map<char, string> obj = {
            {'2', "abc"},
            {'3', "def"},
            {'4', "ghi"},
            {'5', "jkl"},
            {'6', "mno"},
            {'7', "pqrs"},
            {'8', "tuv"},
            {'9', "wxyz"}
        };
        if(digits.empty()) {
            return {};
        }

        vector<string> res;
        string path;

        function<void(int)> backtrack = [&](int start) {
            if(path.size() == digits.size()) {
                res.push_back(path);
                return;
            }
            string letters = obj[digits[start]];
            for(auto letter : letters) {
                path += letter;
                backtrack(start + 1);
                path.pop_back();
            }
        };
        backtrack(0);
        return res;
    }
};
